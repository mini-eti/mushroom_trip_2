
#include <iostream>
bool isPyth(unsigned a, unsigned b, unsigned c) {
	bool x = a * a == (b * b + c * c);
	x = x || c * c == (a * a + b * b);
	x = x || b * b == (c * c + a * a);
	return x;
}

int main()
{
	using u_t = unsigned;
	u_t a = 0, b = 0, c = 0;
	std::cin >> c >> b;
	size_t count = 0;
	while (std::cin >> a) {
		count += isPyth(a, b, c) ? 1 : 0;
		c = b;
		b = a;
	}
	if (std::cin.eof()) {
		std::cout << count;
		std::cout << "\n";
	}
	else if (std::cin.fail()) {
		std::cout << "error\n";
		return 1;
	}
}
